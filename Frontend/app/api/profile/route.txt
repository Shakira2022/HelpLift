import { NextResponse } from "next/server";
import { supabase } from "@/lib/supabase";

export async function GET(req: Request) {
  try {
    const { searchParams } = new URL(req.url);
    const userId = searchParams.get("userId");

    if (!userId) {
      return NextResponse.json({ success: false, message: "User ID is required" }, { status: 400 });
    }

    const { data, error } = await supabase
      .from("users")
      .select(`
        id, full_name, email, phone, role, university, 
        gender, bursary, preferred_area, profile_pic_url, 
        cover_pic_url, created_at
      `)
      .eq("id", userId)
      .single();

    if (error || !data) {
      return NextResponse.json({ success: false, message: "User not found" }, { status: 404 });
    }

    // --- FIXED URL GENERATOR ---
    const getPublicUrl = (value: string | null) => {
      if (!value || value === "null") return null;

      // Already a full URL? Return it
      if (value.startsWith("http")) return value;

      // If it's a filename, encode spaces and special characters
      const fileName = value.includes("/") ? value.split("/").pop() : value;
      const folderPath = `profile-pics/${encodeURIComponent(fileName!)}`;

      const { data: urlData } = supabase.storage
        .from("UniHive")
        .getPublicUrl(folderPath);

      return urlData.publicUrl;
    };

    return NextResponse.json({
      success: true,
      user: {
        id: data.id,
        name: data.full_name,
        email: data.email,
        phoneNumber: data.phone,
        university: data.university,
        gender: data.gender,
        bursary: data.bursary,
        preferredArea: data.preferred_area,
        profilePic: getPublicUrl(data.profile_pic_url),
        coverPic: getPublicUrl(data.cover_pic_url),
        joinDate: new Date(data.created_at).toLocaleDateString("en-ZA", {
          month: "long",
          year: "numeric",
        }),
      },
    });
  } catch (err) {
    console.error("Profile GET error:", err);
    return NextResponse.json({ success: false, message: "Server error" }, { status: 500 });
  }
}


// --- UPDATE USER PROFILE (including profilePic upload URL) ---
export async function PUT(req: Request) {
  try {
    const body = await req.json();
    const {
      id,
      name,
      email,
      phoneNumber,
      university,
      gender,
      bursary,
      preferredArea,
      profilePic,
      coverPic, // <-- new field: public URL of uploaded profile pic
    } = body;

    if (!id) {
      return NextResponse.json(
        { success: false, message: "User ID is required" },
        { status: 400 }
      );
    }

    // Build the update object dynamically
    const updateData: Record<string, any> = {
      full_name: name,
      email,
      phone: phoneNumber,
      university,
      gender,
      bursary,
      preferred_area: preferredArea,
    };

    if (profilePic) {
      updateData.profile_pic_url = profilePic; // save the URL to DB
    }

    if (coverPic) {
      updateData.cover_pic_url = coverPic; 
    }

    const { data, error } = await supabase
      .from("users")
      .update(updateData)
      .eq("id", id)
      .select()
      .single();

    if (error) {
      return NextResponse.json(
        { success: false, message: error.message },
        { status: 500 }
      );
    }

    return NextResponse.json({ success: true, user: data });
  } catch (err) {
    console.error("Profile update error:", err);
    return NextResponse.json(
      { success: false, message: "Server error" },
      { status: 500 }
    );
  }
}
