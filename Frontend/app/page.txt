                         <label className="text-xs font-bold uppercase tracking-widest text-muted-foreground">Direct Email</label>
                         <Input type="email" placeholder="your.email@university.edu" className="bg-background border-border" value={email} onChange={(e) => setEmail(e.target.value)} required />
                      </div>
                      <div className="space-y-2">
                         <Textarea placeholder="Tell me what you're looking for..." className="bg-background border-border min-h-[100px]" value={message} onChange={(e) => setMessage(e.target.value)} required />
                      </div>
                      <Button type="submit" className="w-full gap-2 font-bold">
                         <Send className="w-4 h-4" /> Send Message
                      </Button>
                   </form>
                </div>
             </Card>
          </div>
        </section>

        {/* --- FAQ SECTION --- */}
        <section className="max-w-4xl mx-auto px-4 mt-16 space-y-8">
           <div className="text-center space-y-4">
              <h2 className="text-3xl font-extrabold tracking-tight">Frequently Asked Questions</h2>
              <p className="text-muted-foreground">Everything you need to know about ResiFinder.</p>
           </div>

           {/* FAQ Search Bar */}
           <div className="relative max-w-lg mx-auto">
              <Search className="absolute left-4 top-1/2 -translate-y-1/2 w-5 h-5 text-muted-foreground" />
              <Input 
                type="text" 
                placeholder="Search FAQs..." 
                className="pl-12 h-14 bg-card border-border rounded-2xl shadow-sm focus:ring-primary focus:border-primary"
                value={faqSearch}
                onChange={(e) => setFaqSearch(e.target.value)}
              />
           </div>

           {/* Scrollable Container for FAQs */}
           <div className="max-h-[600px] overflow-y-auto pr-2 custom-scrollbar space-y-3">
              {filteredFaqs.length > 0 ? (
                filteredFaqs.slice(0, visibleFaqCount).map((faq, index) => (
                  <div 
                    key={index} 
                    className="bg-card border border-border rounded-2xl overflow-hidden transition-all duration-300"
                  >
                    <button 
                      onClick={() => setOpenFaq(openFaq === index ? null : index)}
                      className="w-full p-6 flex items-center justify-between text-left hover:bg-muted/50 transition-colors"
                    >
                      <span className="font-bold text-lg pr-4">{faq.question}</span>
                      <div className="bg-primary/10 p-2 rounded-full text-primary">
                        {openFaq === index ? <Minus className="w-4 h-4" /> : <Plus className="w-4 h-4" />}
                      </div>
                    </button>
                    {openFaq === index && (
                      <div className="p-6 pt-0 text-muted-foreground leading-relaxed animate-in slide-in-from-top-2 duration-300">
                         <div className="h-px bg-border mb-4" />
                         {faq.answer}
                      </div>
                    )}
                  </div>
                ))
              ) : (
                <div className="text-center p-10 bg-muted/20 rounded-2xl border border-dashed border-border">
                   <p className="text-muted-foreground font-medium">No FAQs found matching your search.</p>
                </div>
              )}
           </div>

           {/* FAQ Controls */}
           {!faqSearch && (
             <div className="flex flex-col sm:flex-row items-center justify-center gap-4 pt-4">
                {visibleFaqCount < filteredFaqs.length && (
                  <Button 
                    onClick={showMoreFaqs} 
                    variant="outline" 
                    className="w-full sm:w-auto gap-2 border-primary/20 hover:bg-primary/5 font-bold"
                  >
                    <ChevronDown className="w-4 h-4" /> Show More (5)
                  </Button>
                )}
                {visibleFaqCount > 5 && (
                  <Button 
                    onClick={showLessFaqs} 
                    variant="ghost" 
                    className="w-full sm:w-auto gap-2 text-muted-foreground font-bold"
                  >
                    <ChevronUp className="w-4 h-4" /> Collapse Last 5
                  </Button>
                )}
             </div>
           )}
        </section>

        {/* --- FOOTER --- */}
        <footer className="border-t border-border mt-16 py-16 bg-muted/10">
          <div className="max-w-7xl mx-auto px-8 grid grid-cols-1 md:grid-cols-3 gap-12">
            <div className="space-y-4">
              <div className="flex items-center gap-2">
                <Globe className="w-5 h-5 text-primary" />
                <span className="font-bold text-lg">ResiFinder</span>
              </div>
              <p className="text-sm text-muted-foreground">
                The most trusted student accommodation platform for higher education. 
                Helping thousands of students find their perfect home every year.
              </p>
            </div>
            <div className="space-y-4">
              <h4 className="font-bold">Quick Links</h4>
              <div className="flex flex-col gap-2 text-sm text-muted-foreground">
                <a href="#" className="hover:text-primary transition-colors">Safety Guidelines</a>
                <a href="#" className="hover:text-primary transition-colors">Landlord Portal</a>
                <a href="#" className="hover:text-primary transition-colors">Privacy Policy</a>
              </div>
            </div>
            <div className="space-y-4">
              <h4 className="font-bold">Support</h4>
              <p className="text-sm text-muted-foreground">Have questions or need help with your application?</p>
              <Button variant="outline" size="sm" className="w-full">Help Center</Button>
            </div>
          </div>
          <p className="text-center mt-16 text-xs text-muted-foreground/50 italic">© 2024 ResiFinder. Built with ❤️ for the student community.</p>
        </footer>
      </main>
    </div>
  )
}

      </div>
    </div>
  )
}
