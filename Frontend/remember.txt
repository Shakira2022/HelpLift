"use client";

import { useState, useRef, useEffect } from "react";
import { User, X, ThumbsDown, Heart, MessageSquare, MoreVertical, Pencil } from "lucide-react"; // 🔑 ADDED: Pencil

// --- INTERFACES (MODIFIED) ---
export interface CommunityPost {
  id: number;
  author: string;
  category: string;
  title: string;
  content: string;
  likes: number;
  favourites: number;
  // 🔑 MODIFIED: Total post replies count should match the total number of comments
  // The 'replies' property on the post is generally understood as the top-level comment count in this context
  replies: number; 
  timestamp: string;
  images?: string[];
}

export type ReplyType = {
  id: number;
  text: string;
  // 🔑 MODIFIED: Added independent like/dislike counts for replies
  likes: number; 
  dislikes: number;
};

export type CommentType = {
  id: number;
  text: string;
  likes: number;
  dislikes: number;
  replies: ReplyType[];
};

interface CommunityCardProps {
  post: CommunityPost;
  liked?: boolean;
  favourited?: boolean;
  comments: CommentType[]; // Now an initial set of comments
  onLike?: (id: number) => void;
  onFavourite?: (id: number) => void;
  onOpen?: (id: number) => void;
  onImageClick?: (index: number) => void;
  fullHeight?: boolean;
  onCloseFullHeight?: () => void;
  closingPostId?: number | null;
  // 🔑 NEW PROP
  clicked?: boolean; 
  // 🔑 NEW PROPS FOR ACTION MENU
  isAuthor: boolean; // Indicates if the current user is the post author
  onDeletePost: (id: number) => void; // Handler for deleting the post
  // 🔑 NEW PROP ADDED
  onEditPost: (post: CommunityPost) => void; 
}

const INITIAL_REPLIES_VISIBLE = 5;
const REPLIES_INCREMENT = 5;

// Utility function to generate a unique ID for a new reply
const generateUniqueId = () => Math.floor(Math.random() * 1000000);

// Helper to create a unique ID key for state tracking
const getReplyKey = (commentId: number, replyId: number) => `r-${commentId}-${replyId}`;

// 🔑 NEW: Utility function to check if a comment was authored by the current user (using the mock prefix)
const isUserComment = (comment: CommentType) => comment.text.startsWith("(You)");

// 🔑 NEW: Utility function to check if a reply was authored by the current user (using the mock prefix)
const isUserReply = (reply: ReplyType) => reply.text.startsWith("(You)");

// 🔑 NEW: Utility function to check if the user has replied to this parent comment
const hasUserReplied = (comment: CommentType) => comment.replies.some(isUserReply);


export default function CommunityCard({
  post,
  liked = false,
  favourited = false,
  comments: initialComments, // Renamed to initialComments
  onLike,
  onFavourite,
  onOpen,
  onImageClick,
  fullHeight = false,
  onCloseFullHeight,
  closingPostId = null,
  // 🔑 NEW PROP RECEIVE
  clicked = false,
  // 🔑 NEW PROPS RECEIVE
  isAuthor, 
  onDeletePost,
  // 🔑 NEW PROP RECEIVE
  onEditPost,
}: CommunityCardProps) {

  // 🔑 NEW STATE: Local state to manage comments for real-time updates
  const [comments, setComments] = useState<CommentType[]>(initialComments);

  // 🔑 NEW STATE: State for the main post comment input
  const [showNewCommentInput, setShowNewCommentInput] = useState(false);
  const [newCommentText, setNewCommentText] = useState("");

  const [expanded, setExpanded] = useState(false);
  const [showMore, setShowMore] = useState(false);
  const contentRef = useRef<HTMLParagraphElement>(null);
  const commentsRef = useRef<HTMLDivElement>(null);
  const [commentsExpanded, setCommentsExpanded] = useState(false);
  const [slideIn, setSlideIn] = useState(false);

  // 🔑 MODIFIED: State now tracks both comment IDs and reply keys (e.g., 'c-123' or 'r-1-456')
  const [likedItems, setLikedItems] = useState<string[]>([]);
  const [dislikedItems, setDislikedItems] = useState<string[]>([]);

  const [activeReplyCommentId, setActiveReplyCommentId] = useState<number | null>(null);
  const [replyText, setReplyText] = useState<string>(""); // State for reply text input
  const [visibleReplyCounts, setVisibleReplyCounts] = useState<Record<number, number>>({});
  
  // 🔑 NEW STATE: Action Menu State
  const [actionMenuOpen, setActionMenuOpen] = useState(false);
  const actionMenuRef = useRef<HTMLDivElement>(null);


  // --- TOP-LEVEL COMMENT LOGIC (MODIFIED TO USE TOGGLE FUNCTION) ---

  const handleCommentSubmission = (text: string) => {
    if (!text.trim()) return;

    const newComment: CommentType = {
      id: generateUniqueId(),
      text: `(You) ${text}`,
      likes: 0, 
      dislikes: 0,
      replies: [],
    };

    // Prepend the new comment to the list
    setComments(prevComments => [newComment, ...prevComments]);

    // Update the main post's reply count (simulated)
    post.replies += 1; 

    setNewCommentText("");
    setShowNewCommentInput(false);
  };

  const toggleNewCommentInput = () => {
    setNewCommentText("");
    setShowNewCommentInput(prev => {
      // Automatically close the reply input if open when opening the main comment input
      if (!prev && activeReplyCommentId !== null) {
        setActiveReplyCommentId(null);
        setReplyText("");
      }
      return !prev;
    });
  };


  // --- REPLY LOGIC (UNCHANGED) ---

  const handleReplySubmission = (commentId: number, text: string) => {
    if (!text.trim()) return;

    const newReply: ReplyType = {
      id: generateUniqueId(),
      text: `(You) ${text}`,
      // 🔑 NEW: Initialize new replies with zero likes/dislikes
      likes: 0, 
      dislikes: 0,
    };

    setComments(prevComments => {
      return prevComments.map(comment => {
        if (comment.id === commentId) {
          return {
            ...comment,
            replies: [...comment.replies, newReply],
          };
        }
        return comment;
      });
    });

    setVisibleReplyCounts(prevCounts => ({
      ...prevCounts,
      [commentId]: (prevCounts[commentId] || 0) < INITIAL_REPLIES_VISIBLE
        ? INITIAL_REPLIES_VISIBLE
        : (prevCounts[commentId] || 0) + 1,
    }));

    setReplyText("");
    setActiveReplyCommentId(null);
  };

  const toggleReplyForm = (commentId: number) => {
    setReplyText("");
    setActiveReplyCommentId(prevId => {
      // Automatically close the main comment input when opening a reply input
      if (prevId !== commentId && showNewCommentInput) {
        setShowNewCommentInput(false);
        setNewCommentText("");
      }
      return prevId === commentId ? null : commentId;
    });
  };


  // --- REPLY VISIBILITY LOGIC (UNCHANGED) ---

  const getVisibleReplyCount = (commentId: number) => {
    return visibleReplyCounts[commentId] || 0;
  };

  const loadMoreReplies = (commentId: number, totalReplies: number) => {
    setVisibleReplyCounts(prevCounts => {
      const currentCount = prevCounts[commentId] || INITIAL_REPLIES_VISIBLE;
      return {
        ...prevCounts,
        [commentId]: Math.min(currentCount + REPLIES_INCREMENT, totalReplies),
      };
    });
  };

  const toggleReplies = (commentId: number, totalReplies: number) => {
    setVisibleReplyCounts(prevCounts => {
      const currentCount = prevCounts[commentId] || 0;
      const newCount = currentCount === 0 ? Math.min(INITIAL_REPLIES_VISIBLE, totalReplies) : currentCount;

      return {
        ...prevCounts,
        [commentId]: newCount,
      };
    });
  };

  const collapseReplies = (commentId: number) => {
    setVisibleReplyCounts(prevCounts => ({
      ...prevCounts,
      [commentId]: 0,
    }));
  }

  // --- MODIFIED ACTION HANDLERS (Independent Like/Dislike - UNCHANGED) ---

  const handleLike = (id: number, type: 'comment' | 'reply', parentCommentId?: number) => {
    const key = type === 'comment' ? `c-${id}` : getReplyKey(parentCommentId!, id);
    
    setComments(prevComments =>
      prevComments.map(comment => {
        if (type === 'comment' && comment.id === id) {
          const isLiked = likedItems.includes(key);
          const isDisliked = dislikedItems.includes(key);

          let newLikes = comment.likes;
          let newDislikes = comment.dislikes;

          if (isLiked) {
            setLikedItems(prev => prev.filter(k => k !== key));
            newLikes -= 1;
          } else {
            setLikedItems(prev => [...prev, key]);
            newLikes += 1;
            if (isDisliked) {
              setDislikedItems(prev => prev.filter(k => k !== key));
              newDislikes -= 1;
            }
          }
          return { ...comment, likes: newLikes, dislikes: newDislikes };
        } 
        
        // Handle reply like
        if (type === 'reply' && comment.id === parentCommentId) {
          return {
            ...comment,
            replies: comment.replies.map(reply => {
              if (reply.id === id) {
                const isLiked = likedItems.includes(key);
                const isDisliked = dislikedItems.includes(key);

                let newLikes = reply.likes;
                let newDislikes = reply.dislikes;

                if (isLiked) {
                  setLikedItems(prev => prev.filter(k => k !== key));
                  newLikes -= 1;
                } else {
                  setLikedItems(prev => [...prev, key]);
                  newLikes += 1;
                  if (isDisliked) {
                    setDislikedItems(prev => prev.filter(k => k !== key));
                    newDislikes -= 1;
                  }
                }
                return { ...reply, likes: newLikes, dislikes: newDislikes };
              }
              return reply;
            }),
          };
        }
        return comment;
      })
    );
  };


  const handleDislike = (id: number, type: 'comment' | 'reply', parentCommentId?: number) => {
    const key = type === 'comment' ? `c-${id}` : getReplyKey(parentCommentId!, id);

    setComments(prevComments =>
      prevComments.map(comment => {
        if (type === 'comment' && comment.id === id) {
          const isDisliked = dislikedItems.includes(key);
          const isLiked = likedItems.includes(key);

          let newDislikes = comment.dislikes;
          let newLikes = comment.likes;

          if (isDisliked) {
            setDislikedItems(prev => prev.filter(k => k !== key));
            newDislikes -= 1;
          } else {
            setDislikedItems(prev => [...prev, key]);
            newDislikes += 1;
            if (isLiked) {
              setLikedItems(prev => prev.filter(k => k !== key));
              newLikes -= 1;
            }
          }
          // FIX APPLIED HERE: Changed return object to use newDislikes for dislikes count (already fixed in previous step)
          return { ...comment, likes: newLikes, dislikes: newDislikes }; 
        }

        // Handle reply dislike (No change needed here from last revision)
        if (type === 'reply' && comment.id === parentCommentId) {
          return {
            ...comment,
            replies: comment.replies.map(reply => {
              if (reply.id === id) {
                const isDisliked = dislikedItems.includes(key);
                const isLiked = likedItems.includes(key);

                let newDislikes = reply.dislikes;
                let newLikes = reply.likes;

                if (isDisliked) {
                  setDislikedItems(prev => prev.filter(k => k !== key));
                  newDislikes -= 1;
                } else {
                  setDislikedItems(prev => [...prev, key]);
                  newDislikes += 1;
                  if (isLiked) {
                    setLikedItems(prev => prev.filter(k => k !== key));
                    newLikes -= 1;
                  }
                }
                return { ...reply, likes: newLikes, dislikes: newDislikes };
              }
              return reply;
            }),
          };
        }
        return comment;
      })
    );
  };
  
  // Aliases for the main comment actions
  const handleLikeComment = (commentId: number) => handleLike(commentId, 'comment');
  const handleDislikeComment = (commentId: number) => handleDislike(commentId, 'comment');

  // 🔑 NEW HANDLER: Combined edit logic
  const handleEditClick = (e: React.MouseEvent) => {
    e.stopPropagation();
    onEditPost(post);
    setActionMenuOpen(false);
  }

  // 🔑 NEW HANDLER: Combined delete logic
  const handleDeleteClick = (e: React.MouseEvent) => {
    e.stopPropagation();
    onDeletePost(post.id); 
    setActionMenuOpen(false);
  }

  // 🔑 NEW: Close menu if click outside
  useEffect(() => {
    const handleClickOutside = (event: MouseEvent) => {
      if (actionMenuRef.current && !actionMenuRef.current.contains(event.target as Node)) {
        setActionMenuOpen(false);
      }
    };

    document.addEventListener("mousedown", handleClickOutside);
    return () => {
      document.removeEventListener("mousedown", handleClickOutside);
    };
  }, [actionMenuRef]);


  // --- EXISTING EFFECTS (UNCHANGED) ---
  useEffect(() => {
    if (fullHeight) {
      setTimeout(() => setSlideIn(true), 10);
    } else {
      setSlideIn(false);
    }
  }, [fullHeight]);

  useEffect(() => {
    if (fullHeight) {
      document.body.style.overflow = "hidden";
    } else {
      document.body.style.overflow = "";
    }
    return () => {
      document.body.style.overflow = "";
    };
  }, [fullHeight]);

  useEffect(() => {
    if (contentRef.current) {
      const el = contentRef.current;
      // 🔑 MODIFIED: Check if the content height exceeds the height when clamped to 2 lines (approx 3rem)
      // The line-clamp-2 is approx 3rem in height with text-sm (1.25 * 3 = 3.75rem or 60px)
      // Let's use a standard threshold for demonstration, e.g., if scrollHeight is significantly larger than clientHeight 
      // when content is initially rendered without the clamp. Since we are using line-clamp-2 in the unexpanded state,
      // we need to check if the scrollHeight (full content height) is greater than the clamped height.
      // A common way to check if clamping occurred is to compare scrollHeight vs. clientHeight 
      // if the element had no clamp, but since it DOES have a clamp by default, we rely on the 
      // fact that if the full content is longer than 2 lines, showMore should be true.
      // To reliably check this, we'll temporarily remove the class, check the scrollHeight, then reapply.
      // A simple check on initial render is if the scrollHeight is > a certain threshold (e.g. 3 lines worth).
      // For simplicity and to avoid complex measurements, we'll assume content is longer if scrollHeight > clientHeight
      // in the initial state, which is a good-enough heuristic for this component.
      
      // Let's rely on the current logic which *should* work if line-clamp-2 is correctly restricting clientHeight
      // and scrollHeight represents the full content length.
      setShowMore(el.scrollHeight > el.clientHeight); 
    }
  }, [post.content, expanded]); // 🔑 MODIFIED: Added 'expanded' to dependencies to re-evaluate after collapse


  // --- UI COMPONENTS ---
  // 🔑 FIX APPLIED: Simplified initialization to safely default to an empty array if post.images is undefined/null.
  const images = post.images || []; 

  const HeartIcon = ({ filled = false, className = "w-4 h-4" }: { filled?: boolean; className?: string }) => (
    <svg
      className={className}
      fill={filled ? "currentColor" : "none"}
      stroke="currentColor"
      viewBox="0 0 24 24"
    >
      <path
        strokeLinecap="round"
        strokeLinejoin="round"
        strokeWidth={2}
        d="M4.318 6.318a4.5 4.5 0 000 6.364L12 20.364l7.682-7.682a4.5 4.5 0 00-6.364-6.364L12 7.636l-1.318-1.318a4.5 4.5 0 00-6.364 0z"
      />
    </svg>
  );

  const ThumbsDownIcon = ({ className = "w-4 h-4" }: { className?: string }) => (
    <ThumbsDown className={className} />
  );

  // 🔑 MODIFIED: Reply-specific UI to handle independent like/dislike and highlight user replies
  const ReplyItem = ({ reply, parentCommentId }: { reply: ReplyType, parentCommentId: number }) => {
    const likeKey = getReplyKey(parentCommentId, reply.id);
    const isLiked = likedItems.includes(likeKey);
    const isDisliked = dislikedItems.includes(getReplyKey(parentCommentId, reply.id));
    const isUserAuthored = isUserReply(reply); // 🔑 NEW: Check if authored by user
    
    return (
      <div 
        className={`flex items-start gap-2 py-2 border-b border-border/50 last:border-b-0 ${
          isUserAuthored ? 'bg-primary/10 rounded-lg p-2 -mx-2' : '' // 🔑 MODIFIED: Highlight for user's reply
        }`}
      >
        <div className="w-8 h-8 rounded-full bg-accent flex items-center justify-center flex-shrink-0 mt-1">
          <User className="w-4 h-4 text-primary" />
        </div>
        <div className="flex flex-col flex-grow">
          <p className="text-sm font-medium text-foreground leading-snug">
            {reply.text}
            {/* 🔑 NEW: Explicit "You" tag for clarity on replies */}
            {isUserAuthored && (
                <span className="ml-2 px-1 text-xs font-bold text-primary bg-primary/20 rounded">
                    (Your Reply)
                </span>
            )}
          </p>
          <div className="flex gap-4 mt-1 items-center">
            {/* Like Button (Independent) */}
            <button
              onClick={(e) => {
                e.stopPropagation();
                handleLike(reply.id, 'reply', parentCommentId);
              }}
              className={`flex items-center gap-1 text-xs transition-colors ${
                isLiked
                  ? "text-destructive"
                  : "text-muted-foreground hover:text-foreground"
              }`}
            >
              <HeartIcon filled={isLiked} className="w-4 h-4" />
              {reply.likes}
            </button>

            {/* Dislike Button (Independent) */}
            <button
              onClick={(e) => {
                e.stopPropagation();
                handleDislike(reply.id, 'reply', parentCommentId);
              }}
              className={`flex items-center gap-1 text-xs transition-colors ${
                isDisliked
                  ? "text-red-500"
                  : "text-muted-foreground hover:text-foreground"
              }`}
            >
              <ThumbsDownIcon className="w-4 h-4" />
              {reply.dislikes}
            </button>
            
            {/* Reply Button - Opens reply form for the PARENT COMMENT (Prevents nesting) */}
            <button
              onClick={(e) => {
                e.stopPropagation();
                toggleReplyForm(parentCommentId);
              }}
              className={`flex items-center gap-1 text-xs font-medium hover:underline ${
                activeReplyCommentId === parentCommentId ? "text-destructive" : "text-primary"
              }`}
            >
              <MessageSquare className="w-4 h-4" />
              {activeReplyCommentId === parentCommentId ? "Cancel" : "Reply"}
            </button>

          </div>
        </div>
      </div>
    );
  };

  // 🔑 MODIFIED LOGIC: Sort comments to pin user's activity
  const sortedComments = [...comments].sort((a, b) => {
    const isAUserComment = isUserComment(a);
    const isBUserComment = isUserComment(b);
    const isAUserReplyParent = hasUserReplied(a);
    const isBUserReplyParent = hasUserReplied(b);

    // Define priority score for sorting
    const getPriority = (isComment: boolean, isReplyParent: boolean) => {
      if (isComment) return 3; // Highest priority: Comment created by user
      if (isReplyParent) return 2; // Second highest: User replied to this comment
      return 1; // Default priority: No user activity
    };

    const priorityA = getPriority(isAUserComment, isAUserReplyParent);
    const priorityB = getPriority(isBUserComment, isBUserReplyParent);

    if (priorityA > priorityB) return -1; // A (higher priority) comes first
    if (priorityA < priorityB) return 1;  // B (higher priority) comes first
    
    // Tie-breaker: If priorities are equal, sort by ID (newest first)
    return b.id - a.id;
  });
  
  // 🔑 MODIFIED: Check if the current user has submitted any comments or replies for the post-level indicator
  const hasUserCommented = sortedComments.some(c => isUserComment(c) || hasUserReplied(c));


  return (
    <div
      className={
        fullHeight
          ? "fixed inset-0 z-50 bg-background overflow-y-auto w-full h-full active:bg-gray-200/30"
          : "w-full border-b-4 border-border overflow-hidden cursor-pointer hover:shadow-lg transition-all relative active:bg-gray-200/30"
      }
      onClick={() => !expanded && onOpen && onOpen(post.id)}
    >
      <div
        // 🔑 MODIFIED: Conditional class to show 'clicked' state when not fullHeight
        className={`w-full border-b-1 border-border overflow-hidden cursor-pointer transition-all ${
          fullHeight 
            ? "flex-1 overflow-y-auto bg-background" 
            : "hover:shadow-lg"
        } ${
          !fullHeight && clicked ? "bg-muted/30" : "bg-background" // If not full height, apply muted/30 if clicked, otherwise background
        }`}
      >
        {/* HEADER (MODIFIED STRUCTURE) */}
        <div
          className="px-4 pt-4 pb-2 relative"
          onClick={() => !expanded && onOpen && onOpen(post.id)}
        >
          {/* Author, Timestamp, and Action Menu - MODIFIED */}
          <div className="flex items-start justify-between mb-1">
            <div className="flex items-start gap-2">
              <button
                onClick={(e) => e.stopPropagation()}
                className="w-11 h-11 rounded-lg bg-accent flex items-center justify-center hover:bg-accent/70 transition"
              >
                <User className="w-6 h-6 text-muted-foreground" />
              </button>
              {/* 🔑 MODIFIED: Changed flex-col to flex-row and added items-center to align name/timestamp/divider */}
              <div className="flex items-center"> 
                <p className="text-sm font-semibold text-foreground">{post.author}</p>
                
                {/* 🔑 FIX APPLIED (FROM PREV REQUEST): Removed 'hidden sm:block' to always show the divider */}
                <span className="mx-2 h-3 w-px bg-muted-foreground/50" /> 

                {/* 🔑 FIX APPLIED (FROM PREV REQUEST): Removed 'hidden sm:block' to always show the timestamp */}
                <span className="text-xs text-muted-foreground">{post.timestamp}</span>
              </div>
            </div>

            {/* 🔑 MODIFIED: Combined timestamp and action menu container (Timestamp moved to author block) */}
            <div className="flex items-center gap-2"> 
                {/* 🔑 REMOVED: Timestamp moved to be inline with author name */}
                {/* <span className="text-xs text-muted-foreground mr-1 hidden sm:block">{post.timestamp}</span> */}

                <div className="relative" ref={actionMenuRef}>
                    {/* 🔑 NEW: Action Menu Toggle */}
                    <button
                        onClick={(e) => {
                            e.stopPropagation();
                            setActionMenuOpen(prev => !prev);
                        }}
                        className="w-8 h-8 flex items-center justify-center rounded-full text-muted-foreground hover:bg-accent/50 transition"
                        aria-label="Post actions"
                    >
                        <MoreVertical className="w-5 h-5" />
                    </button>

                    {/* 🔑 MODIFIED: Action Menu Dropdown - wider, with close button */}
                    {actionMenuOpen && (
                        <div
                            onClick={(e) => e.stopPropagation()} 
                            // 🔑 MODIFIED WIDTH: w-56, added p-1 for internal padding
                            className="absolute right-0 top-full mt-1 w-56 bg-card border border-border rounded-lg shadow-lg z-30 overflow-hidden p-1" 
                        >
                            {/* 🔑 NEW: Close Button in the Dropdown */}
                            <div className="flex justify-end p-1">
                                <button 
                                    onClick={() => setActionMenuOpen(false)}
                                    className="text-muted-foreground hover:text-foreground p-1 transition-colors"
                                    aria-label="Close menu"
                                >
                                    <X className="w-4 h-4" />
                                </button>
                            </div>
                            
                            {isAuthor ? (
                                // Options for the author
                                <>
                                    {/* 🔑 NEW: Edit Post Button */}
                                    <button
                                        onClick={handleEditClick}
                                        className="w-full flex items-center gap-2 text-left px-3 py-2 text-sm text-primary hover:bg-primary/10 transition-colors rounded-md"
                                    >
                                        <Pencil className="w-4 h-4" />
                                        Edit Post
                                    </button>
                                    <div className="my-1 border-t border-border/50" />
                                    <button
                                        onClick={handleDeleteClick} // Use combined handler
                                        className="w-full flex items-center gap-2 text-left px-3 py-2 text-sm text-destructive hover:bg-destructive/10 transition-colors rounded-md"
                                    >
                                        <X className="w-4 h-4" />
                                        Delete Post
                                    </button>
                                </>
                            ) : (
                                // Options for other users
                                <>
                                    <button
                                        onClick={() => setActionMenuOpen(false)}
                                        className="w-full text-left px-3 py-2 text-sm text-destructive hover:bg-destructive/10 transition-colors rounded-md"
                                    >
                                        Report Post
                                    </button>
                                    <button
                                        onClick={() => setActionMenuOpen(false)}
                                        className="w-full text-left px-3 py-2 text-sm text-muted-foreground hover:bg-accent/50 transition-colors rounded-md"
                                    >
                                        Mute Author (WIP)
                                    </button>
                                </>
                            )}
                        </div>
                    )}
                </div>
            </div>
            {/* END: Action Menu */}
          </div>
  
          {/* Title + Close Button */}
          <div className="flex items-start justify-between mb-1">
            <h3 className="font-semibold text-foreground">{post.title}</h3>
            {fullHeight && onCloseFullHeight && (
              <button
                onClick={(e) => {
                  e.stopPropagation();
                  onCloseFullHeight();
                }}
                className="w-7 h-7 flex items-center justify-center rounded-md border border-border bg-background hover:bg-accent/50 font-bold text-lg transition"
              >
                <X className="w-4 h-4" />
              </button>
            )}
          </div>
  
          <span className="text-xs font-medium text-primary">#{post.category}</span>
        </div>
  
        {/* IMAGES (UNCHANGED) */}
        {images.length > 0 && (
          <div className="relative w-full h-40 flex gap-1">
            {images.length === 1 && (
              <img
                src={images[0]}
                alt={post.title}
                className="w-full h-full object-cover"
                onClick={(e) => {
                  e.stopPropagation();
                  onImageClick && onImageClick(0);
                }}
              />
            )}
            {images.length === 2 && (
              <>
                <img
                  src={images[0]}
                  alt=""
                  className="w-1/2 h-full object-cover"
                  onClick={(e) => {
                    e.stopPropagation();
                    onImageClick && onImageClick(0);
                  }}
                />
                <img
                  src={images[1]}
                  alt=""
                  className="w-1/2 h-full object-cover"
                  onClick={(e) => {
                    e.stopPropagation();
                    onImageClick && onImageClick(1);
                  }}
                />
              </>
            )}
            {images.length > 2 && (
              <>
                <div className="relative w-1/2 h-full">
                  <img
                    src={images[0]}
                    alt=""
                    className="w-full h-full object-cover"
                    onClick={(e) => {
                      e.stopPropagation();
                      onImageClick && onImageClick(0);
                    }}
                  />
                  <div
                    className="absolute inset-0 flex items-center justify-center bg-black/40 text-white text-lg font-bold cursor-pointer"
                    onClick={(e) => {
                      e.stopPropagation();
                      onImageClick && onImageClick(0);
                    }}
                  >
                    +{images.length - 1}
                  </div>
                </div>
                <img
                  src={images[1]}
                  alt=""
                  className="w-1/2 h-full object-cover"
                  onClick={(e) => {
                    e.stopPropagation();
                    onImageClick && onImageClick(1);
                  }}
                />
              </>
            )}
          </div>
        )}
  
        {/* CONTENT (MODIFIED LOGIC) */}
        <div className="px-4 py-3">
          <p
            ref={contentRef}
            className={`text-sm text-foreground ${expanded ? "" : "line-clamp-2"}`}
          >
            {post.content}
          </p>
          
          <div className="flex justify-between items-center mt-1">
            {/* 🔑 MODIFIED: Show 'View More' only if NOT expanded AND if content is actually truncated */}
            {!expanded && showMore && (
              <button
                onClick={(e) => {
                  e.stopPropagation();
                  setExpanded(true);
                }}
                className="text-xs text-primary font-medium hover:underline"
              >
                ...View More
              </button>
            )}
            
            {/* 🔑 NEW: Collapse Button - Show only if expanded */}
            {expanded && (
              <button
                onClick={(e) => {
                  e.stopPropagation();
                  setExpanded(false); // Collapse the content
                  // Since showMore is re-evaluated on 'expanded' change, 
                  // the 'View More' button will reappear if content is still truncated.
                }}
                className="text-xs text-destructive font-medium hover:underline ml-auto" // Added ml-auto to push it right
              >
                View Less (Collapse)
              </button>
            )}
          </div>
        </div>
  
        {/* ACTIONS (Styling Unchanged) */}
        <div className="flex items-center gap-6 px-4 py-3 border-t border-border mx-4">
          {/* 1. Like */}
          <button
            onClick={(e) => {
              e.stopPropagation();
              onLike && onLike(post.id);
            }}
            className={`flex items-center gap-1 text-sm transition-colors ${
              liked ? "text-destructive" : "text-muted-foreground hover:text-foreground"
            }`}
          >
            <HeartIcon filled={liked} className="w-5 h-5" />
            {post.likes}
          </button>
  
          {/* 2. Comments Icon (FIX APPLIED HERE) */}
          <button
            onClick={(e) => {
              e.stopPropagation();
              // 🔑 FIX APPLIED: Removed the `!expanded &&` condition so the comment button works
              // even after the user has clicked '...View More' to fully expand the text content.
              if (onOpen) { 
                onOpen(post.id);
              }
            }}
            className="flex items-center gap-1 text-sm text-muted-foreground hover:text-foreground transition-colors"
          >
            <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                strokeWidth={2}
                d="M8 12h.01M12 12h.01M16 12h.01M21 12c0 4.418-4.03 8-9 8a9.863 9.863 0 01-4.255-.949L3 20l1.395-3.72C3.512 15.042 3 13.574 3 12c0-4.418 4.03-8 9-8s9 3.582 9 8z"
              />
            </svg>
            {comments.length}
            {/* 🔑 NEW: Indicator for user activity */}
            {hasUserCommented && (
                <span className="ml-1 text-xs text-primary font-bold">
                    *YOU*
                </span>
            )}
          </button>
  
          {/* 3. Favourite */}
          <button
            onClick={(e) => {
              e.stopPropagation();
              onFavourite && onFavourite(post.id);
            }}
            className={`flex items-center gap-1 text-sm transition-colors ${
              favourited ? "text-primary" : "text-muted-foreground hover:text-foreground"
            }`}
          >
            <svg
              className="w-5 h-5"
              fill={favourited ? "currentColor" : "none"}
              stroke="currentColor"
              viewBox="0 0 24 24"
            >
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                strokeWidth={2}
                d="M5 5a2 2 0 012-2h10a2 2 0 012 2v16l-7-3.5L5 21V5z"
              />
            </svg>
            {post.favourites}
          </button>

          {/* 4. NEW: Text Comment Button (Now always visible) */}
          <button
            onClick={(e) => {
              e.stopPropagation();
              // If the card is not in full view, open it first.
              if (!fullHeight && onOpen) {
                onOpen(post.id);
                // Use a slight delay to ensure the state updates (fullHeight = true) before toggling the input.
                // This ensures the input appears inside the fullHeight comments section.
                setTimeout(() => toggleNewCommentInput(), 0); 
              } else {
                 toggleNewCommentInput();
              }
            }}
            className={`flex items-center gap-1 text-sm font-medium transition-colors hover:underline ${
              showNewCommentInput && fullHeight ? "text-destructive" : "text-primary"
            }`}
          >
            {showNewCommentInput && fullHeight ? "Cancel Comment" : "Comment"}
          </button>

        </div>
  
        {/* COMMENTS SECTION (Only shown in fullHeight mode) */}
        {fullHeight && (
          <div
            ref={commentsRef}
            className={`
              px-4 w-full h-[60vh] overflow-y-auto border-t-4 border-border space-y-2
              bg-background
              transition-transform duration-300 ease-in-out
              ${closingPostId === post.id
                ? "translate-x-full opacity-100"
                : slideIn
                ? "translate-x-0 opacity-100"
                : "-translate-x-full opacity-0"
              }
            `}
          >
            <div className="sticky top-0 bg-background z-10 py-2">
              <h4 className="font-semibold text-sm text-foreground">
                Comments - {comments.length}
              </h4>
  
              <button
                onClick={(e) => {
                  e.stopPropagation();
                  const el = commentsRef.current;
                  if (!el) return;
  
                  setCommentsExpanded(true);
                  el.scrollIntoView({ behavior: "smooth", block: "start" });
                }}
                className="absolute right-2 top-2 w-7 h-7 flex items-center justify-center rounded-md border border-border bg-background hover:bg-accent/50 transition"
              >
                <span className="absolute inset-0 flex items-center justify-center text-base leading-none">v</span>
                <span className="absolute inset-0 flex items-center justify-center text-xs leading-none">v</span>
              </button>
  
              {comments.length > 0 && (
                <p className="text-sm text-muted-foreground mt-1 border-b border-border">
                  {comments.length} comment{comments.length > 1 ? "s" : ""}
                </p>
              )}
            </div>

            {/* TOP-LEVEL COMMENT INPUT */}
            {showNewCommentInput && (
              <div className="mt-3 flex items-start gap-2 p-3 border border-border bg-card rounded-lg shadow-inner sticky top-16 z-20">
                <div className="w-10 h-10 rounded-full bg-accent flex items-center justify-center flex-shrink-0">
                  <User className="w-5 h-5 text-primary" />
                </div>
                <div className="flex flex-col w-full">
                  <textarea
                    className="w-full text-sm resize-none border border-border p-2 rounded-md focus:outline-none focus:ring-1 focus:ring-primary bg-background/80 text-foreground placeholder:text-muted-foreground"
                    rows={3}
                    placeholder="Write a top-level comment..."
                    value={newCommentText}
                    onChange={(e) => setNewCommentText(e.target.value)}
                    onClick={(e) => e.stopPropagation()}
                  />
                  <div className="flex justify-end gap-2 mt-1">
                    <button
                      onClick={(e) => {
                        e.stopPropagation();
                        toggleNewCommentInput();
                      }}
                      className="text-xs font-medium text-muted-foreground hover:text-foreground"
                    >
                      Cancel
                    </button>
                    <button
                      onClick={(e) => {
                        e.stopPropagation();
                        handleCommentSubmission(newCommentText);
                      }}
                      disabled={!newCommentText.trim()}
                      className="text-sm font-semibold px-4 py-1 bg-primary text-primary-foreground rounded-md hover:bg-primary/90 transition disabled:bg-muted-foreground disabled:cursor-not-allowed"
                    >
                      Post Comment
                    </button>
                  </div>
                </div>
              </div>
            )}
            {/* END: TOP-LEVEL COMMENT INPUT */}
  
            {comments.length > 0 ? (
              <>
                {/* 🔑 MODIFIED: Map over sortedComments instead of comments */}
                {sortedComments.map((comment) => {
                  const totalReplies = comment.replies.length;
                  const currentVisibleCount = getVisibleReplyCount(comment.id);
                  const visibleReplies = comment.replies.slice(0, currentVisibleCount);
                  const remainingRepliesCount = totalReplies - visibleReplies.length;
                  const isRepliesCurrentlyVisible = currentVisibleCount > 0;
                  const shouldShowViewMore =
                    remainingRepliesCount > 0 &&
                    currentVisibleCount > 0 &&
                    currentVisibleCount < totalReplies;
                  const shouldShowCollapseButton = isRepliesCurrentlyVisible;

                  const commentKey = `c-${comment.id}`;
                  const isCommentLiked = likedItems.includes(commentKey);
                  const isCommentDisliked = dislikedItems.includes(commentKey);
                  
                  const isUserAuthored = isUserComment(comment); // Check if authored by user
                  const isUserParent = hasUserReplied(comment); // Check if user replied to it
                  
                  // Highlight based on activity: user-comment takes precedence
                  const highlightClass = isUserAuthored 
                    ? 'border-2 border-primary ring-2 ring-primary/50' 
                    : isUserParent
                    ? 'border-2 border-accent ring-1 ring-accent/50' 
                    : '';

                  // Indicator text based on activity:
                  const indicatorText = isUserAuthored 
                    ? "(Your Comment)" 
                    : isUserParent 
                    ? "(User Replied)" 
                    : null;
  
                  return (
                    <div 
                      key={comment.id} 
                      className={`border border-border p-3 rounded-lg bg-card/60 ${highlightClass}`} // 🔑 MODIFIED: Dynamic highlight based on activity
                    >
                      <p className="text-sm text-foreground">
                        {comment.text}
                        {/* 🔑 MODIFIED: Explicit indicator tag */}
                        {indicatorText && (
                          <span className={`ml-2 px-1 text-xs font-bold rounded ${
                            isUserAuthored ? 'text-primary bg-primary/20' : 'text-accent-foreground bg-accent/20'
                          }`}>
                            {indicatorText}
                          </span>
                        )}
                      </p>
  
                      {/* Comment Actions */}
                      <div className="flex gap-4 mt-1 items-center">
                        {/* Like Button */}
                        <button
                          onClick={(e) => {
                            e.stopPropagation();
                            handleLikeComment(comment.id);
                          }}
                          className={`flex items-center gap-1 text-xs transition-colors ${
                            isCommentLiked
                              ? "text-destructive"
                              : "text-muted-foreground hover:text-foreground"
                          }`}
                        >
                          <HeartIcon filled={isCommentLiked} />
                          {comment.likes}
                        </button>
  
                        {/* Dislike Button */}
                        <button
                          onClick={(e) => {
                            e.stopPropagation();
                            handleDislikeComment(comment.id);
                          }}
                          className={`flex items-center gap-1 text-xs transition-colors ${
                            isCommentDisliked
                              ? "text-red-500"
                              : "text-muted-foreground hover:text-foreground"
                          }`}
                        >
                          <ThumbsDownIcon className="w-4 h-4" />
                          {comment.dislikes}
                        </button>
  
                        {/* Replies Button (Toggle visibility of nested replies) */}
                        <button
                          onClick={(e) => {
                            e.stopPropagation();
                            toggleReplies(comment.id, totalReplies);
                          }}
                          className={`flex items-center gap-1 text-xs transition-colors font-medium ${
                            isRepliesCurrentlyVisible && totalReplies > 0
                              ? "text-primary"
                              : "text-muted-foreground hover:text-foreground"
                          }`}
                        >
                          <MessageSquare className="w-4 h-4" />
                          {`Replies (${totalReplies})`}
                        </button>
  
                        {/* Reply Button (Opens reply form for THIS comment) */}
                        <button
                          onClick={(e) => {
                            e.stopPropagation();
                            toggleReplyForm(comment.id);
                          }}
                          className={`flex items-center gap-1 text-xs font-medium hover:underline ${
                            activeReplyCommentId === comment.id ? "text-destructive" : "text-primary"
                          }`}
                        >
                          {activeReplyCommentId === comment.id ? "Cancel" : "Reply"}
                        </button>
                      </div>
  
                      {/* Reply Input */}
                      {activeReplyCommentId === comment.id && (
                        <div className="mt-3 flex items-start gap-2 p-3 border border-border bg-card rounded-lg shadow-inner">
                          <div className="w-8 h-8 rounded-full bg-accent flex items-center justify-center flex-shrink-0">
                            <User className="w-4 h-4 text-primary" />
                          </div>
                          <div className="flex flex-col w-full">
                            <textarea
                              className="w-full text-sm resize-none border border-border p-2 rounded-md focus:outline-none focus:ring-1 focus:ring-primary bg-background/80 text-foreground placeholder:text-muted-foreground"
                              rows={2}
                              placeholder={`Replying to comment ${comment.id}...`}
                              value={replyText}
                              onChange={(e) => setReplyText(e.target.value)}
                              onClick={(e) => e.stopPropagation()}
                            />
                            <div className="flex justify-end gap-2 mt-1">
                              <button
                                onClick={(e) => {
                                  e.stopPropagation();
                                  toggleReplyForm(comment.id);
                                }}
                                className="text-xs font-medium text-muted-foreground hover:text-foreground"
                              >
                                Cancel
                              </button>
                              <button
                                onClick={(e) => {
                                  e.stopPropagation();
                                  handleReplySubmission(comment.id, replyText);
                                }}
                                disabled={!replyText.trim()}
                                className="text-xs font-semibold px-3 py-1 bg-primary text-primary-foreground rounded-md hover:bg-primary/90 transition disabled:bg-muted-foreground disabled:cursor-not-allowed"
                              >
                                Post Reply
                              </button>
                            </div>
                          </div>
                        </div>
                      )}
  
                      {/* Replies Container - MODIFIED TO USE ReplyItem component */}
                      {totalReplies > 0 && visibleReplies.length > 0 && (
                        <div className="ml-4 mt-2 space-y-1 border-l-2 border-border pl-2">
                          {visibleReplies.map((reply) => (
                            // 🔑 MODIFIED: Using the new ReplyItem component
                            <ReplyItem 
                              key={reply.id} 
                              reply={reply} 
                              parentCommentId={comment.id}
                            />
                          ))}
  
                          {/* View More / Collapse */}
                          {(shouldShowViewMore || shouldShowCollapseButton) && (
                            <div className={`flex items-center pt-1 ${shouldShowViewMore ? "justify-between" : "justify-end"}`}>
                              {shouldShowViewMore && (
                                <button
                                  onClick={(e) => {
                                    e.stopPropagation();
                                    loadMoreReplies(comment.id, totalReplies);
                                  }}
                                  className="text-xs text-primary font-medium hover:underline"
                                >
                                  View {Math.min(remainingRepliesCount, REPLIES_INCREMENT)} more replies...
                                </button>
                              )}
  
                              {shouldShowCollapseButton && (
                                <button
                                  onClick={(e) => {
                                    e.stopPropagation();
                                    collapseReplies(comment.id);
                                  }}
                                  className={`text-xs text-primary font-medium hover:underline ${shouldShowViewMore ? "" : "ml-auto"}`}
                                >
                                  Collapse replies
                                </button>
                              )}
                            </div>
                          )}
                        </div>
                      )}
                    </div>
                  );
                })}
              </>
            ) : (
              <p className="text-sm text-muted-foreground mt-2">No Comments</p>
            )}
          </div>
        )}
      </div>
    </div>
  );
}
